200 inline void thread_pool_worker::run_loop(std::stop_token token)
201 {
202 while (!token.stop_requested() && !pool->shutdown.load())
203 {
204 std::function<void()> task;
205 bool has_task = false;
206
207 {
208 std::unique_lock<std::mutex> lock(pool->mutex);
209
210 auto status = pool->cv.wait_for(lock, pool->get_idle_timeout(), [this, &token]()
211 { return !pool->tasks.empty() || token.stop_requested() || pool->shutdown.load(); });
212
213 if (!status && pool->tasks.empty() && !pool->shutdown.load())
214 {
215 // Timeout occurred and no tasks available
216 if (pool->workers.size() > pool->get_min_threads())
217 {
218 // Mark this worker for removal instead of removing immediately
219 auto worker = std::move(pool->workers.at(get_id()));
220 pool->workers.erase(get_id());
221 pool->workers_to_remove.push_back(std::move(worker));
222 pool->available_workers.fetch_sub(1);
223 return;
224 }
225 }
226
227 if (token.stop_requested() || pool->shutdown.load())
228 {
229 pool->available_workers.fetch_sub(1);
230 return;
231 }
232
233 if (!pool->tasks.empty())
234 {
235 task = std::move(pool->tasks.front());
236 pool->tasks.pop();
237 has_task = true;
238 }
239 }
240 pool->available_workers.fetch_sub(1);
241
242 // Execute task with exception handling
243 if (has_task)
244 {
245 try
246 {
247 task();
248 }
249 catch (...)
250 {
251 // Log exception or handle appropriately
252 // For now, we'll just continue to prevent thread termination
253 }
254 }
255 pool->available_workers.fetch_add(1);
256 }
