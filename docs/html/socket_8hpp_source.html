240 void shutdown_channel(socket_stream_mode mode)
241 {
242 if (!descriptor)
243 throw std::logic_error("Descriptor is null");
244 if (mode == socket_stream_mode::READ && !read_shutdown)
245 {
246 descriptor->shutdown(socket_stream_mode::READ);
247 this->read_shutdown = true;
248 }
249 else if (mode == socket_stream_mode::WRITE && !write_shutdown)
250 {
251 descriptor->shutdown(socket_stream_mode::WRITE);
252 this->write_shutdown = true;
253 }
254 }
