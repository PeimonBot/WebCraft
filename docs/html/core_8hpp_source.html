58 {

125 template <typename R>
126 async_readable_stream<R> auto to_readable_stream(async_generator<R> &&gen)
127 {
128 struct stream
129 {
130 async_generator<R> gen;
131 std::optional<typename async_generator<R>::iterator> it;
132
133 explicit stream(async_generator<R> &&gen) : gen(std::move(gen)) {}
134
135 task<std::optional<R>> recv()
136 {
137 if (!it.has_value())
138 {
139 it = co_await gen.begin();
140 }
141
142 if (it == gen.end())
143 {
144 co_return std::nullopt;
145 }
146
147 auto &itr = *it;
148 std::optional<R> value = std::move(*itr);
149 co_await ++itr;
150 co_return std::move(value);
151 }
152 };
153
154 static_assert(async_readable_stream<stream, R>, "Should be an async_readable_stream");
155
156 return stream{std::move(gen)};

167 task<std::optional<T>> get_next()

191 co_return co_await awaitable{*this};

195 {
196 values.push(val);

199 continuation.resume();
