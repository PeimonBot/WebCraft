232 template <typename T>
233 struct reducer_collector
234 {
235 std::function<T(T, T)> func;
236
237 explicit reducer_collector(std::function<T(T, T)> &&f)
238 : func(std::move(f)) {}
239
240 task<T> operator()(async_generator<T> gen) const
241 {
242 auto itr = co_await gen.begin();
243
244 if (itr == gen.end())
245 {
246 throw std::runtime_error("Cannot reduce an empty generator");
247 }
248 T result = *itr;
249 co_await ++itr;
250
251 while (itr != gen.end())
252 {
253 result = func(std::move(result), *itr);
254 co_await ++itr;
255 }
256
257 co_return result;
258 }
259 };
